-------------------MainActivity.java
package com.example.listview221025;

import android.content.Intent;
import android.os.Bundle;
import android.util.Log;
import android.view.View;
import android.widget.AdapterView;
import android.widget.ListView;
import android.widget.Toast;

import androidx.activity.EdgeToEdge;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity implements AdapterView.OnItemClickListener {

    private static final String TAG = "ListviewDebug";
    private ListView vistita;

    // *** Asegúrate que los 3 arreglos tengan 3 elementos ***
    String arreglo1[] = {"uno", "dos", "tres"};
    String arreglo2[] = {"deportes 1", "deportes 2", "deportes 3"};
    int imagenes[] = {
            android.R.drawable.ic_menu_directions,
            android.R.drawable.ic_menu_directions,
            android.R.drawable.ic_menu_directions
    };

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        EdgeToEdge.enable(this);
        setContentView(R.layout.activity_main);

        vistita = findViewById(R.id.vistita);

        Adaptador adaptador = new Adaptador(
                getApplicationContext(),
                arreglo1, arreglo2, imagenes
        );
        vistita.setAdapter(adaptador);
        vistita.setOnItemClickListener(this);

        // ====== CONFIRMACIÓN: deben ser 3 ======
        int count = adaptador.getCount();
        Log.d(TAG, "getCount() = " + count
                + " | a1=" + arreglo1.length
                + " a2=" + arreglo2.length
                + " imgs=" + imagenes.length);
        Toast.makeText(this, "items = " + count, Toast.LENGTH_SHORT).show();
        // =======================================
    }

    @Override
    public void onItemClick(AdapterView<?> parent, View view, int position, long id) {
        if (position == 0) {
            startActivity(new Intent(this, DetalleActivity.class));
        } else {
            Toast.makeText(this, "Elemento " + arreglo1[position], Toast.LENGTH_SHORT).show();
        }
    }
}
------------------Adaptador.java
package com.example.listview221025;

import android.content.Context;
import android.util.Log;
import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;
import android.widget.BaseAdapter;
import android.widget.ImageView;
import android.widget.TextView;

public class Adaptador extends BaseAdapter {

    private static final String TAG = "ListviewDebug";

    Context context;
    String lista1[];
    String lista2[];
    int listaimagenes[];
    LayoutInflater inflater;

    public Adaptador(Context context, String[] lista1, String[] lista2, int[] listaimagenes) {
        this.context = context;
        this.lista1 = lista1;
        this.lista2 = lista2;
        this.listaimagenes = listaimagenes;
        this.inflater = LayoutInflater.from(context);

        Log.d(TAG, "Constructor: l1=" + (lista1==null?0:lista1.length)
                + " l2=" + (lista2==null?0:lista2.length)
                + " imgs=" + (listaimagenes==null?0:listaimagenes.length));
    }

    @Override
    public int getCount() {
        int n1 = (lista1 != null) ? lista1.length : 0;
        int n2 = (lista2 != null) ? lista2.length : 0;
        int n3 = (listaimagenes != null) ? listaimagenes.length : 0;
        int min = Math.min(n1, Math.min(n2, n3));
        Log.d(TAG, "getCount -> min(" + n1 + "," + n2 + "," + n3 + ") = " + min);
        return min;
    }

    @Override public Object getItem(int position) { return lista1[position]; }
    @Override public long getItemId(int position) { return position; }

    @Override
    public View getView(int position, View convertView, ViewGroup parent) {
        if (convertView == null)
            convertView = inflater.inflate(R.layout.basurita, parent, false);

        TextView textview = convertView.findViewById(R.id.textview);
        TextView textview2 = convertView.findViewById(R.id.textview2);
        ImageView imageicon = convertView.findViewById(R.id.imageicon);

        textview.setText(lista1[position]);
        textview2.setText(lista2[position]);
        imageicon.setImageResource(listaimagenes[position]);

        return convertView;
    }
}
-------------------DetalleActivity.java
package com.example.listview221025;

import android.os.Bundle;
import android.view.Menu;
import android.view.MenuItem;
import android.view.View;
import android.widget.ImageView;

import androidx.activity.EdgeToEdge;
import androidx.annotation.NonNull;
import androidx.appcompat.app.AppCompatActivity;
import androidx.fragment.app.Fragment;

import com.example.listview221025.R;

public class DetalleActivity extends AppCompatActivity {

    private ImageView ivPalomita;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        EdgeToEdge.enable(this); // si te da lata, puedes borrar esta línea
        setContentView(R.layout.activity_detalle);

        ivPalomita = findViewById(R.id.iv_palomita);   // <-- asegúrate que exista en el XML
        ivPalomita.setOnClickListener(v -> mostrarFragment(new HolaFragment()));
    }

    private void mostrarFragment(Fragment f){
        View cont = findViewById(R.id.fl_contenedor_detalle); // <-- asegúrate que exista en el XML
        cont.setVisibility(View.VISIBLE);
        getSupportFragmentManager()
                .beginTransaction()
                .replace(R.id.fl_contenedor_detalle, f)
                .commit();
    }

    @Override
    public boolean onCreateOptionsMenu(Menu menu) {
        getMenuInflater().inflate(R.menu.mainmenu, menu);
        return true;
    }

    @Override
    public boolean onOptionsItemSelected(@NonNull MenuItem item) {
        if (item.getItemId() == R.id.opcion_unica) {
            mostrarFragment(new HolaFragment());
            return true;
        }
        return super.onOptionsItemSelected(item);
    }
}

package com.example.listview221025;

import android.os.Bundle;
import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;
import android.widget.Button;
import android.widget.Toast;

import androidx.annotation.NonNull;
import androidx.annotation.Nullable;
import androidx.fragment.app.Fragment;

import com.example.listview221025.R;

public class HolaFragment extends Fragment {

    public HolaFragment(){}

    @Nullable
    @Override
    public View onCreateView(@NonNull LayoutInflater inflater,
                             @Nullable ViewGroup container,
                             @Nullable Bundle savedInstanceState) {
        return inflater.inflate(R.layout.fragment_hola, container, false);
    }

    @Override
    public void onViewCreated(@NonNull View v, @Nullable Bundle s) {
        Button btnSaludito = v.findViewById(R.id.btn_saludito);
        btnSaludito.setOnClickListener(view ->
                Toast.makeText(requireContext(), "hola", Toast.LENGTH_SHORT).show());
    }
}
